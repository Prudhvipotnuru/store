	pid int,
	pname varchar(100),
	price int,
	image varchar(500),
	hsn int
);


create table os_customer(
	cid int,
	cname varchar(50),
	cmobilbe varchar(20),
	clocation varchar(20)
)

create table os_orders(
	oid int,
	odate Date,
	ototal int,
	cid int
);


create table os_order_products(
	oid int,
	pid int
);


create table os_hsn(
	hsn_code int,
	gst int
);

create table os_product_category(
	pid int,
	category_name varchar(100)
);
